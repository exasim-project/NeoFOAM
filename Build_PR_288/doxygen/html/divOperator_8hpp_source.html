32 std::string key = (std::holds_alternative<Dictionary>(inputs))

35 DivOperatorFactory<ValueType>::keyExistsOrError(key);

121
122 DivOperator(
123 dsl::Operator::Type termType,
124 const SurfaceField<scalar>& faceFlux,

126 )

128 faceFlux_(faceFlux), divOperatorStrategy_(nullptr) {};
129
130
131 void explicitOperation(Field<scalar>& source) const
132 {
133 NF_ASSERT(divOperatorStrategy_, "DivOperatorStrategy not initialized");
134 NeoFOAM::Field<NeoFOAM::scalar> tmpsource(source.exec(), source.size(), 0.0);
135 const auto operatorScaling = this->getCoefficient();
136 divOperatorStrategy_->div(tmpsource, faceFlux_, this->getField(), operatorScaling);
137 source += tmpsource;
138 }
139
140 la::LinearSystem<ValueType, localIdx> createEmptyLinearSystem() const

156 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField(), operatorScaling);
157 }
158
159 void div(la::LinearSystem<ValueType, localIdx>& ls) const

162 divOperatorStrategy_->div(ls, faceFlux_, this->getField(), operatorScaling);
163 };
164
165 void div(VolumeField<ValueType>& divPhi) const
166 {
167 const auto operatorScaling = this->getCoefficient();
168 divOperatorStrategy_->div(divPhi, faceFlux_, this->getField(), operatorScaling);
169 }
170
171 void build(const Input& input)
172 {
173 const UnstructuredMesh& mesh = this->getField().mesh();
174 if (std::holds_alternative<NeoFOAM::Dictionary>(input))
175 {
176 auto dict = std::get<NeoFOAM::Dictionary>(input);
177 std::string schemeName = "div(" + faceFlux_.name + "," + this->getField().name + ")";
178 auto tokens = dict.subDict("divSchemes").get<NeoFOAM::TokenList>(schemeName);
179 divOperatorStrategy_ =
180 DivOperatorFactory<ValueType>::create(this->exec(), mesh, tokens);
181 }
182 else
183 {
184 auto tokens = std::get<NeoFOAM::TokenList>(input);
185 divOperatorStrategy_ =
186 DivOperatorFactory<ValueType>::create(this->exec(), mesh, tokens);
187 }
188 }
189
190 std::string getName() const { return "DivOperator"; }
191
192private:
193
194 const SurfaceField<NeoFOAM::scalar>& faceFlux_;
195
196 std::unique_ptr<DivOperatorFactory<ValueType>> divOperatorStrategy_;
197};
