30/* @brief solve an expression
31 *
32 * @param exp - Expression which is to be solved/updated.
33 * @param solution - Solution field, where the solution will be 'written to'.
34 * @param t - the time at the start of the time step.
35 * @param dt - time step for the temporal integration
36 * @param fvSchemes - Dictionary containing spatial operator and time integration properties
37 * @param fvSolution - Dictionary containing linear solver properties
38 */
39template<typename FieldType>
40void solve(
41 Expression<typename FieldType::ElementType>& exp,
42 FieldType& solution,
43 scalar t,
44 scalar dt,
45 [[maybe_unused]] const Dictionary& fvSchemes,
46 [[maybe_unused]] const Dictionary& fvSolution
47)
48{
49 // TODO:
50 if (exp.temporalOperators().size() == 0 && exp.spatialOperators().size() == 0)
51 {
52 NF_ERROR_EXIT("No temporal or implicit terms to solve.");
53 }
54 exp.build(fvSchemes);
55 if (exp.temporalOperators().size() > 0)
56 {
57 // integrate equations in time
58 timeIntegration::TimeIntegration<FieldType> timeIntegrator(
59 fvSchemes.subDict("ddtSchemes"), fvSolution
60 );
61 timeIntegrator.solve(exp, solution, t, dt);
62 }
63 else
64 {
65 // solve sparse matrix system
66 using ValueType = typename FieldType::ElementType;
67
68 auto sparsity = NeoN::finiteVolume::cellCentred::SparsityPattern(solution.mesh());
69 auto ls = la::createEmptyLinearSystem<
70 ValueType,
71 localIdx,
72 NeoN::finiteVolume::cellCentred::SparsityPattern>(sparsity);
73
74 exp.implicitOperation(ls);
75 auto expTmp = exp.explicitOperation(solution.mesh().nCells());
76
77 auto [vol, expSource, rhs] = spans(solution.mesh().cellVolumes(), expTmp, ls.rhs());
78
79 // subtract the explicit source term from the rhs
80 parallelFor(
81 solution.exec(),
82 {0, rhs.size()},
83 KOKKOS_LAMBDA(const size_t i) { rhs[i] -= expSource[i] * vol[i]; }
84 );
85
86#if NF_WITH_GINKGO
87 auto solver = la::ginkgo::Solver<ValueType>(solution.exec(), fvSolution);
88 solver.solve(ls, solution.internalField());
89#else
90 NF_ERROR_EXIT("No linear solver is available, build with -DNeoN_WITH_GINKGO=ON");
91#endif
92 }
93}
