22void setFixedValue(
23 DomainField<ValueType>& domainField, std::pair<size_t, size_t> range, ValueType fixedValue
24)
25{
26 auto [refGradient, value, valueFraction, refValue] = spans(
27 domainField.boundaryField().refGrad(),
28 domainField.boundaryField().value(),
29 domainField.boundaryField().valueFraction(),
30 domainField.boundaryField().refValue()
31 );
32
33 NeoN::parallelFor(
34 domainField.exec(),
35 range,
36 KOKKOS_LAMBDA(const size_t i) {
37 refValue[i] = fixedValue;
38 value[i] = fixedValue;
39 valueFraction[i] = 1.0; // only used refValue
40 refGradient[i] = fixedValue; // not used
41 }
42 );
43}
