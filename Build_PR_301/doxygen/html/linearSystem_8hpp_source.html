102template<typename ValueType, typename IndexType>
103Field<ValueType> spmv(LinearSystem<ValueType, IndexType>& ls, Field<ValueType>& xfield)
104{
105 Field<ValueType> resultField(ls.exec(), ls.rhs().size(), 0.0);
106 auto [result, b, x] = spans(resultField, ls.rhs(), xfield);
107
108 auto values = ls.matrix().values().span();
109 auto colIdxs = ls.matrix().colIdxs().span();
110 auto rowPtrs = ls.matrix().rowPtrs().span();
111
112 parallelFor(
113 ls.exec(),
114 {0, ls.matrix().nRows()},
115 KOKKOS_LAMBDA(const std::size_t rowi) {
116 IndexType rowStart = rowPtrs[rowi];
117 IndexType rowEnd = rowPtrs[rowi + 1];
118 ValueType sum = 0.0;
119 for (IndexType coli = rowStart; coli < rowEnd; coli++)
120 {
121 sum += values[coli] * x[colIdxs[coli]];
122 }
123 result[rowi] = sum - b[rowi];
124 }
125 );
126
127 return resultField;
128};

133convertLinearSystem(const LinearSystem<ValueTypeIn, IndexTypeIn>& ls)
134{
135 auto exec = ls.exec();
136 Field<ValueTypeOut> convertedRhs(exec, ls.rhs().data(), ls.rhs().size());

148LinearSystem<ValueType, IndexType> createEmptyLinearSystem(const SparsityType& sparsity)
149{
150 const auto& exec = sparsity.mesh().exec();

152 localIdx rows {sparsity.rows()};

156 CSRMatrix<ValueType, IndexType> {

159 Field<ValueType> {exec, rows, zero<ValueType>()}
