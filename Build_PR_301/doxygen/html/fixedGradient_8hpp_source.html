22template<typename ValueType>
23void setGradientValue(
24 DomainField<ValueType>& domainField,
25 const UnstructuredMesh& mesh,
26 std::pair<size_t, size_t> range,
27 ValueType fixedGradient
28)
29{
30 const auto iField = domainField.internalField().span();
31
32 auto [refGradient, value, valueFraction, refValue, faceCells, deltaCoeffs] = spans(
33 domainField.boundaryField().refGrad(),
34 domainField.boundaryField().value(),
35 domainField.boundaryField().valueFraction(),
36 domainField.boundaryField().refValue(),
37 mesh.boundaryMesh().faceCells(),
38 mesh.boundaryMesh().deltaCoeffs()
39 );
40
41 NeoN::parallelFor(
42 domainField.exec(),
43 range,
44 KOKKOS_LAMBDA(const size_t i) {
45 refGradient[i] = fixedGradient;
46 // operator / is not defined for all ValueTypes
47 value[i] =
48 iField[static_cast<size_t>(faceCells[i])] + fixedGradient * (1 / deltaCoeffs[i]);
49 valueFraction[i] = 0.0; // only used refGrad
50 refValue[i] = fixedGradient; // not used
51 },
52 "setGradientValue"
53 );
54}
55}
